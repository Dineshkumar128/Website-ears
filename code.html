Snippet 01 — sw.js
Service Worker: Pre-caching all assets on install
The Service Worker listens for the install event and uses waitUntil() to defer activation until every asset is safely in cache. This guarantees offline availability from the very first visit.
PWA · OFFLINE
// Cache version — bump to 'ears-v3' to force fresh install
const CACHE_NAME = 'ears-v2';

const urlsToCache = [
  './', './index.html', './script.js',
  './manifest.json', './data.json',
  './fonts/dm-sans-400.ttf'   // fonts cached locally — CDN not needed offline
];

self.addEventListener('install', event => {
  event.waitUntil(                         // ← A: blocks activation until done
    caches.open(CACHE_NAME)
      .then(cache => cache.addAll(urlsToCache))
  );
});

self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request)
      .then(response =>
        response || fetch(event.request)  // ← B: cache hit instant, miss → network
      )
  );
});
AwaitUntil() ensures worker doesn't activate until every file is cached — atomic install, no partial state
BCache-first strategy: offline users get instant responses; online users still benefit from zero-latency cache hits
CChanging CACHE_NAME to 'ears-v3' forces all clients to re-download assets on next visit — safe versioning
Snippet 02 — script.js
The DB abstraction — localStorage with a clean API
All data access routes through a single DB object. This is the key architectural seam that makes future database migration trivial — replace this object, touch nothing else.
ARCHITECTURE · DATA
// All data access goes through here
// To swap to a real DB: replace only this object
const DB = {
  get:    k => JSON.parse(localStorage.getItem('ears3_' + k) || 'null'),
  set:    (k, v) => localStorage.setItem('ears3_' + k, JSON.stringify(v)),
  remove: k => localStorage.removeItem('ears3_' + k),  // ← A: namespaced prefix

  // Typed accessors — always return array, never null
  acts:      () => DB.get('acts')  || [],  // ← B: safe defaults
  saveActs:  v  => DB.set('acts', v),
  users:     () => DB.get('users') || [],
  saveUsers: v  => DB.set('users', v),
  audit:     () => DB.get('audit') || [],
  saveAudit: v  => DB.set('audit', v),

  // Auto-incrementing ID — mimics a database SERIAL sequence
  nextId: key => {                             // ← C: per-collection counters
    const value = (DB.get(key) || 0) + 1;
    DB.set(key, value);
    return value;
  }
};
A'ears3_' namespace prefix prevents key collisions with other apps sharing the same origin's localStorage
B|| [] default means consumers never null-check — contract is always an array, reducing defensive code throughout
CnextId() is a self-contained counter per collection — equivalent to PostgreSQL SERIAL type, with zero dependencies
Snippet 03 — script.js
Password validation + XSS sanitisation
Every password is tested against five independent rules. The esc() function HTML-encodes all four dangerous metacharacters on every dynamic output — verified by external security review.
SECURITY · AUTH
function validatePassword(pw) {
  if (pw.length < 8)         return 'Must be at least 8 characters.';
  if (!/[A-Z]/.test(pw))      return 'Needs at least one uppercase letter.';
  if (!/[a-z]/.test(pw))      return 'Needs at least one lowercase letter.';
  if (!/[0-9]/.test(pw))      return 'Needs at least one number.';
  if (!/[^A-Za-z0-9]/.test(pw)) return 'Needs a special character.'; // ← A
  return null;  // null = valid — callers check for non-null string
}

// XSS guard — every dynamic string passes through this
function esc(value) {                            // ← B
  return String(value || '')
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');  // ← C: 4 metacharacters covered
}
ANull-return pattern — returns a string on failure, null on success. Clean contract with no thrown exceptions anywhere in the chain
Besc() is called on every user-controlled value before innerHTML — verified by external reviewer to block injection
CEncoding &, <, >, " covers the four HTML injection vectors — covers attribute injection, tag injection, and entity abuse
Snippet 04 — script.js
Google Sheets sync — production-ready, dormant until configured
The full two-way sync pipeline. Three guard clauses make it impossible to fire if not configured, concurrent, or offline. The finally block always releases the lock — even on throw.
FUTURE-READY · SYNC
async function syncToGoogleSheets(manual = false) {
  const SCRIPT_URL = DB.get('googleScriptUrl');
  if (!SCRIPT_URL)      return;  // ← A: guard 1 — not configured
  if (syncInProgress)   return;  // ← A: guard 2 — prevent overlap
  if (!navigator.onLine) return; // ← A: guard 3 — offline check

  syncInProgress = true;
  try {
    const toSync = DB.acts().filter(a =>
      !SYNC_CONFIG.excludeStatuses.includes(a.status)
    );
    const response = await fetch(SCRIPT_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({           // ← B: incremental sync payload
        activities:   toSync,
        lastSyncTime: lastSyncTime,
        clientId:     'ears-' + Date.now()
      })
    });
    if (response.ok) {
      lastSyncTime = nowIso();
      DB.set('lastSyncTime', lastSyncTime);
    }
  } catch (error) {
    if (manual) toast('Sync failed', 'error');
  } finally {
    syncInProgress = false;  // ← C: always release lock
  }
}
AThree guard clauses exit immediately with zero side effects — impossible to fire accidentally in any state
BlastSyncTime enables incremental sync — only records changed since last sync are sent, reducing payload size
Cfinally block always releases syncInProgress — prevents a stuck lock if fetch throws an unexpected error
Snippet 05 — script.js
boot() — the single application entry point
Everything starts here. Seeds data, restores session, registers the Service Worker, and routes to the correct page based on auth state and role — all in one clean function.
INIT · PWA
function boot() {
  seedData();          // ← A: idempotent — only runs on true first visit
  updateAuthUI();

  const savedConfig = DB.get('syncConfig');
  if (savedConfig) Object.assign(SYNC_CONFIG, savedConfig);
  lastSyncTime = DB.get('lastSyncTime');

  // Role-based routing — B
  if (isLoggedIn()) {
    showPage(isAdmin() ? 'admin' : 'dashboard');
  } else {
    showLoginGate();
  }

  // Progressive enhancement — C
  if ('serviceWorker' in navigator) {
    navigator.serviceWorker.register('./sw.js')
      .then(() => {})
      .catch(() => {});  // app still works without SW
  }
}

boot();  // called immediately on page load
AseedData() uses 'usersSeeded' / 'actsSeeded' flags — demo data only inserted on first run, never overwrites real data
BRole-based routing in one line: admins → admin panel, users → dashboard, unauthenticated → login gate
CProgressive enhancement — 'serviceWorker' in navigator means the app works perfectly on browsers without SW support
